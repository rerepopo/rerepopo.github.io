 73    def buttonClicked(self):
 74        text = self.ui.lineEdit.text()
 75        # ref https://vimsky.com/zh-tw/examples/detail/python-method-PyQt4.QtGui.QFont.html
 76        palettePass = QtGui.QPalette()
 77        palettePass.setColor(QtGui.QPalette.Text, QtCore.Qt.black)

 87        try:
 88            if text[0:2] == "0x":  # if hex
 89                intVal = int(text[2:], 16)
 90                binVal = bin(intVal)
 91            elif text[0:2] == "0b":  # if bin
 92                intVal = int(text[2:], 2)
 93                binVal = bin(intVal)
 94            else:  # if dec
 95                intVal = int(text, 10)
 96                binVal = bin(intVal)
 97            #print(" binVal is " + binVal)

 99            if intVal > (pow(2,32)-1):
100                intVal = (pow(2,32)-1)
101                binVal = bin(intVal)
102                self.ui.lineEdit.setPalette(paletteFail)
103                self.ui.lineEdit.setFont(fontFail)
104                self.ui.labelInputCheckResult.setText(str(text) + ' is invalid input. Take it as 0xFFFFFFFF.  \nNOTE: Only Support 0~2^32-1.')
105            elif intVal < 0:
106                intVal = 0
107                binVal = bin(intVal)
108                self.ui.lineEdit.setPalette(paletteFail)
109                self.ui.lineEdit.setFont(fontFail)
110                self.ui.labelInputCheckResult.setText(str(text)+' is invalid input. Take it as 0. \nNOTE: Only Support 0~2^32-1.')
111            else:
112                self.ui.lineEdit.setPalette(palettePass)
113                self.ui.lineEdit.setFont(fontPass)
114                self.ui.labelInputCheckResult.setText("Input Check Result: PASS.")
115                print("Input Check Result: PASS.")
116            self.inputIntValue = intVal
117        except:
118            intVal = self.inputIntValue
119            binVal = bin(intVal)
120            self.inputIntValue = self.inputIntValue
121            self.ui.lineEdit.setPalette(paletteFail)
122            self.ui.lineEdit.setFont(fontFail)
123            self.ui.labelInputCheckResult.setText(str(text)+' is invalid input. \nNOTE: "^0x[0-9A-Fa-f]+$" for hex. "^0b[0-1]+$" for binary. "^[0-9]+$" for decimal.')
124                 #    /^       # start of string
125                 #    0x       # 0x prefix
126                 #    [0-9A-F] # a hex digit
127                 #    $        # end of string
128        # for used:
129        index = -1
130        while binVal[index] in ["1", "0"]:
131            item = self.ui.tableWidget.item(0, 32 + index)
132            item.setText(binVal[index])
133            index = index - 1
